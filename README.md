Detach default styles example with [Stitches](https://stitches.dev/).

```ts
import { destyle } from 'destyle-ts'

// Stitches
import { globalCss } from '@stitches/react'

// Root Component

	return (

			{initStyle()}
			<YOUR_COMPONENT />
		</>
	)
}
```
